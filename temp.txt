local wezterm = require("wezterm")

local M = {}

function M.apply(config)
	-- Appearance
	config.adjust_window_size_when_changing_font_size = false
	config.enable_scroll_bar = false

	config.font = wezterm.font("JetBrainsMono Nerd Font Mono", { weight = "Regular" }) -- 'Thin' can look choppy on older screens
	config.font_size = 14 -- Smaller font for performance and readability on older displays
	config.line_height = 1.0
	config.cell_width = 1.0

	config.force_reverse_video_cursor = false
	config.freetype_load_flags = "NO_HINTING"
	config.freetype_render_target = "Normal" -- Use simpler rendering for older GPUs

	config.inactive_pane_hsb = {
		saturation = 1.0,
		brightness = 1.0,
	}

	config.window_background_opacity = 1.0
	config.window_decorations = "RESIZE" -- Avoid transparency-related decorations
	config.window_padding = { left = 2, right = 2, top = 2, bottom = 2 }

	-- Performance
	config.term = "xterm-256color"
	config.front_end = "Software" -- Best choice for machines with poor GPU acceleration
	config.prefer_egl = false -- Disable EGL if GPU drivers are weak or buggy
	config.webgpu_power_preference = "LowPower" -- Let the system use integrated GPU sensibly
	config.max_fps = 60 -- Lower FPS to reduce GPU stress
	config.underline_thickness = "1pt"

	-- Workspace
	config.default_workspace = "HOME"
end

return M
